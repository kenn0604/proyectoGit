git init "crea un archivo git en el proyecto"
git add "agrea los archivos modificados a subir"
git commit -m "agrega los cambios al historial de git con una descripción de lo realizado"
git remote add origin "enlaza el archivo original con el repositorio de github"
git push origin "sube los cambios al main o a la rama en la que se esté trabajando"
git clone "crea una copia local del repositorio"
git pull origin main "baja o descarga los cambios del main o una rama en específico"
git checkout -b "crea una rama"
git cheackout "cambiar entre ramas"
