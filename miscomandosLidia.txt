git init "crea un archivo git en el proyecto"
git add "agrega los archivos modificados a subir"
git commit -m "agrega los cambios al historial de git con una descripcion de lo realizado"
git remote add origin "enlaza el archivo original con el repositorio de gitub"
git push origin "sube los cambios al main o a la rama en la que se este trabajando"
git clone "crea una copia local del repositorio"
